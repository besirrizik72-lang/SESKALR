function startAR() { const base = "https://raw.githubusercontent.com/besirrizik72-lang/SESKALR/main/"; // Google'a 3 dosyayı birden tanıtan özel komut dizini const model1 = encodeURIComponent(base + "2LIKOLTUK_RDTAND.glb"); const model2 = encodeURIComponent(base + "4BITERKOLTUK.glb"); const model3 = encodeURIComponent(base + "BITTERSEHPA.glb"); // 'link' parametresi ile birden fazla objeyi aynı sahnede zorluyoruz const intent = `intent://arvr.google.com/scene-viewer/1.0?file=${model1}&link=${model2}&link=${model3}&mode=ar_only#Intent;scheme=https;package=com.google.android.googlequicksearchbox;action=android.intent.action.VIEW;end;`; window.location.href = intent; }
